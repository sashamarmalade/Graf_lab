#include <iostream>
using namespace std;

int main()
{
	setlocale(0, "");
	int n;
	cout << "Введите количество чисел, которое вы хотите ввести: ";
	cin >> n;
	cout << "Введите " << n << " чисел: " << endl;
	int* a = new int[n];    // создаем динамический массив
	for (int i = 0; i < n; i++)
	{
		cin >> a[i];  // считываем числа в ячейки массива
	}

	cout << " Вывод четных чисел массива в упорядоченном по возрастанию порядке: ";

	for (int i = 0; i < n; i++) 
	{
		cout << a[i] << " ";  // выводим значение всех ячеек
	}

	return 0;
}
