#include <ctime>
#include <iostream>
#include <windows.h>
#include "stdio.h"

using namespace std;


void StrSum(char* str3, char str1[255],  int n, int size)
{
	for (int i = 0; i < n; i++)
	{
		str3[i] = str1[i];
	}

	str3[n] = ' ';
	n++;

	str3[size + 1] = '\0';
}

void main()
{
	setlocale(LC_ALL, "ru");
	srand(time(0));

	char str1[255];

	cout << "Введите строку 1: "; gets_s(str1);

	int n = 0;
	while (true)
	{
		if (str1[n] == '\0') break;
		n++;
	}
	cout << endl << "Количество символов str1: " << n << endl;


	int size = n;
	char* str3 = new char[size + 1];

	StrSum(str3, str1, n, size);

	cout << endl << str3 << endl;

}
