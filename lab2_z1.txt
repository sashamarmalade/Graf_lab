#include "iostream"
using namespace std;

int main()
{
	setlocale(LC_ALL, "Russian");
	int A, B, N;
	    cout << "Размер матрицы:" << endl;
		cout << "N=" ; cin >> N;
		cout << "Верхняя граница:"; cin >> A;
	    cout << "Нижняя граница:" ; cin >>B;
		int **a = new int *[N];
		for (int i = 0; i < N; i++)
			a[i] = new int[N];
		for (int i = 0; i < N; i++)
			for (int j = 0; j < N; j++)
				a[i][j] = A + rand() % ((B + 1) - A);
		cout << "Массив:" << endl;
		for (int i = 0; i < N; i++)
		{
			cout << " "<<endl;
			for (int j = 0; j < N; j++)
			{
				cout.width(3);
				cout << a[i][j] << " ";
			}
			cout << endl;
		}
		cout << endl;
		for (int i = 0; i < N; i++)
			delete[] a[i];
		delete[] a;
}
