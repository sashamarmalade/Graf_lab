#include <iostream>
#include <ctime>
using namespace std;

int main()
{
	setlocale(0, "");
	srand((unsigned int)time(0));
    cout << "Введите размер массива:" << endl;
	int n;//размер массива
	cin >> n;
	int* a = new int[n];

	for (int i = 0; i < n; ++i)
	{
		a[i] = rand() % 50;
	}

	cout << "Исходный массив:" << endl;
	for (int i = 0; i < n; ++i)
	{
		cout << a[i] << " ";
	}
	cout << endl;

	cout << "Четные элементы:" << endl;
	for (int i = 0; i < n; ++i)
	{
		if (!(a[i] & 1))
			cout << a[i] << endl;
	}
	for (int i = 0; i < n - 1; i++)
		for (int j = i + 1; j < n; j++)
			if (a[j] < a[i])
			{
				int t = a[j];
				a[j] = a[i];
				a[i] = t;
			}

	for (int i = 0; i < n; i++)
		cout << a[i] << " ";
	cout << endl;
	system("pause>0");
	return 0;
}
