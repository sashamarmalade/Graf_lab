#include <iostream>
#include <ctime>
#include <fstream>
using namespace std;

int main()
{
	setlocale(0, "");
	int A, B, N;
	cout << "Введите A:" << endl;	cin >> A;
	cout << "Введите B:" << endl;	cin >> B;
	cout << "Введите размер массива:" << endl; cin >> N;
	int* a = new int[N];
	cout << "Исходный массив:" << endl;
	for (int i = 0; i < N; i++) {
a[i] = A + rand() % ((B + 1) - A);
cout << " " << a[i];
	}
	cout << endl;
	int ch = 0;
	int n = 0;
	int nechet = 0;
//кол-во четных элементов 
	while (n < N) {
		if (a[n] % 2 == 0) {
			ch += 1;
		}
		n++;
	}
	cout << "Количество четных элементов:" << ch <<endl;
	//сжимаем массив
	nechet = N - ch;
	int* b = new int[N - nechet];
	int now = 0;
	for (int i = 0; i < N; i++)
	{
		if (a[i] % 2 == 0) // проверяем значение на четность
		{
			b[now++] = a[i];  // если четное, то помещаем в новый массив число Х          
		}
	
	}
	cout << "Новый массив:";
	for (int i = 0; i < now; i++)
		cout << b[i] << " ";
	cout << endl;
	return 0;
}
