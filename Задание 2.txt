#include <iostream>
#include <ctime>
#include <fstream>
using namespace std;

int main()
{
	setlocale(0, "");
	int a, b;
	int n, j, i;
	cout << "Введите a:" << endl;	cin >> a;
	cout << "Введите b:" << endl;	cin >> b;
	cout << "Введите размер массива:" << endl; cin >> n;
	int* A = new int[n];
	for (int i = 0; i < n; i++)
		A[i] = a + rand() % ((b + 1) - a);

	cout << "Исходный массив:" << endl;
	for (int i = 0; i < n; ++i)
	{
		cout << A[i] << " ";
	}

	for (i = 0; i < n; i++) cout <<" ";
	cout << endl;
	j = 0;
	for (i = 0; i < n; i++)
		if ((A[i] % 2) == 0)
			A[j++] = A[i];
	n = j;
	cout << "Четные элементы массива:" << endl;
	for (i = 0; i < n; i++) cout << A[i] << " ";
	cout << endl;
	return 0;

}
