#include <iostream>
#include <ctime>
using namespace std;

int max = 0;

int maximum(int* A, int i)
{
	if (i == 1) return A[1];
	else
		if (A[i] > maximum(A, i - 1)) return A[i];
				
				else return maximum(A,i-1);

	
}


int main()
{
	setlocale(LC_ALL, "ru");
	srand(time(NULL));
	int size;
	int i = 0;

	cout << "Введите размер массива: "; 
	cin >> size;

	int* A = new int[size];

	for (int i = 0; i < size; i++)
	{
		A[i] = rand() % 100; // [-100, 100]
		printf("A[%d] = %d\n", i, A[i]);
	}

	printf("Максимальное значение элементов массива: A[%d] = %d\n", size, maximum(A, i-1));

	delete[] A;

	return 0;
}
